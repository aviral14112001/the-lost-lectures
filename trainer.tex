\chapter{Switching from C to C++}
\section{Why so serious?}

\begin{verbatim}
#include <stdio.h>

int main() {

        int n;
        long long m;
        char s[1000];

        scanf("%d%lld%s", &n, &m, &s);

        printf("%d %lld %s\n", n, s);

        return 0;
}
\end{verbatim}

becomes

\begin{verbatim}
#include <bits/stdc++.h>
using namespace std;

int main() {

        int n;
        long long m;
        char s[1000];

        string ss;

        cin >> n >> m >> s >> ss;

        cout << n << ' ' << m << ' ' << s << ' ' << ss << '\n';

        return 0;
}
\end{verbatim}
